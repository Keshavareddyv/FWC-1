%================ QUESTIONS =================%
\begin{enumerate}[label=\textbf{\arabic*.}, leftmargin=*]

\item Degree of the differential equation  
$\sin x + \cos\!\left(\dfrac{dy}{dx}\right) = y^2$  
is
\begin{enumerate}[label=(\Alph*)]
\item $2$
\item $1$
\item not defined
\item $0$
\end{enumerate}

\item The integrating factor of the differential equation  
$(1-y^2)\dfrac{dx}{dy} + yx = ay,\ (-1<y<1)$  
is
\begin{enumerate}[label=(\Alph*)]
\item $\dfrac{1}{y^2-1}$
\item $\dfrac{1}{\sqrt{y^2-1}}$
\item $\dfrac{1}{1-y^2}$
\item $\dfrac{1}{\sqrt{1-y^2}}$
\end{enumerate}

\item Unit vector along $\vec{PQ}$, where  
$P(2,1,-1)$ and $Q(4,4,-7)$  
is
\begin{enumerate}[label=(\Alph*)]
\item $2\hat{i}+3\hat{j}-6\hat{k}$
\item $-2\hat{i}-3\hat{j}+6\hat{k}$
\item $\dfrac{-2\hat{i}}{7}-\dfrac{3\hat{j}}{7}+\dfrac{6\hat{k}}{7}$
\item $\dfrac{2\hat{i}}{7}+\dfrac{3\hat{j}}{7}-\dfrac{6\hat{k}}{7}$
\end{enumerate}

\item If in $\triangle ABC$, $\vec{BA}=2\vec{a}$ and $\vec{BC}=3\vec{b}$, then $\vec{AC}$ is
\begin{enumerate}[label=(\Alph*)]
\item $2\vec{a}+3\vec{b}$
\item $2\vec{a}-3\vec{b}$
\item $3\vec{b}-2\vec{a}$
\item $-2\vec{a}-3\vec{b}$
\end{enumerate}

\item If $|\vec{a}\times\vec{b}|=\sqrt{3}$ and $\vec{a}\cdot\vec{b}=-3$, then the angle between $\vec{a}$ and $\vec{b}$ is
\begin{enumerate}[label=(\Alph*)]
\item $\dfrac{2\pi}{3}$
\item $\dfrac{\pi}{6}$
\item $\dfrac{\pi}{3}$
\item $\dfrac{5\pi}{6}$
\end{enumerate}

\item Equation of the line passing through origin and making  
$30^\circ, 60^\circ, 90^\circ$ with $x,y,z$ axes respectively is
\begin{enumerate}[label=(\Alph*)]
\item $\dfrac{2x}{\sqrt{3}}=\dfrac{y}{2}=\dfrac{z}{0}$
\item $\dfrac{2x}{\sqrt{3}}=\dfrac{2y}{1}=\dfrac{z}{0}$
\item $2x=\dfrac{2y}{\sqrt{3}}=\dfrac{z}{1}$
\item $\dfrac{2x}{\sqrt{3}}=\dfrac{2y}{1}=\dfrac{z}{1}$
\end{enumerate}

\item If $P(A|B)=2P(B|A)$ and $P(A)+P(B)=\dfrac{2}{3}$, then $P(B)$ is
\begin{enumerate}[label=(\Alph*)]
\item $\dfrac{2}{9}$
\item $\dfrac{7}{9}$
\item $\dfrac{4}{9}$
\item $\dfrac{5}{9}$
\end{enumerate}

\item Anti-derivative of  
$\dfrac{\tan x - 1}{\tan x + 1}$  
is
\begin{enumerate}[label=(\Alph*)]
\item $\sec^2\!\left(\dfrac{\pi}{4}-x\right)+c$
\item $-\sec^2\!\left(\dfrac{\pi}{4}-x\right)+c$
\item $\log\!\left|\sec\!\left(\dfrac{\pi}{4}-x\right)\right|+c$
\item $-\log\!\left|\sec\!\left(\dfrac{\pi}{4}-x\right)\right|+c$
\end{enumerate}

\item If $(a,b),(c,d),(e,f)$ are vertices of $\triangle ABC$ and $\Delta$ is its area, then  
\[
\left|
\begin{pmatrix}
a & c & e \\
b & d & f \\
1 & 1 & 1
\end{pmatrix}
\right|^2
\]
is equal to
\begin{enumerate}[label=(\Alph*)]
\item $2\Delta^2$
\item $4\Delta^2$
\item $2\Delta$
\item $4\Delta$
\end{enumerate}

\item The function $f(x)=x|x|$ is
\begin{enumerate}[label=(\Alph*)]
\item continuous and differentiable at $x=0$
\item continuous but not differentiable at $x=0$
\item differentiable but not continuous at $x=0$
\item neither differentiable nor continuous at $x=0$
\end{enumerate}

\end{enumerate}
