\begin{enumerate}[label=\arabic*.]

\item Degree of the differential equation  
$\sin x + \cos\left(\dfrac{dy}{dx}\right) = y^2$ is
\begin{enumerate}[label=(\Alph*)]
\item $2$
\item $1$
\item not defined
\item $0$
\end{enumerate}

\item The integrating factor of the differential equation  
$(1-y^2)\dfrac{dx}{dy} + yx = ay,\; (-1 < y < 1)$ is
\begin{enumerate}[label=(\Alph*)]
\item $1/(y^2-1)$
\item $1/\sqrt{y^2-1}$
\item $1/(1-y^2)$
\item $1/\sqrt{1-y^2}$
\end{enumerate}

\item Unit vector along $PQ$, where $P(2,1,-1)$ and $Q(4,4,-7)$ is
\begin{enumerate}[label=(\Alph*)]
\item $2i + 3j - 6k$
\item $-2i - 3j + 6k$
\item $-2i/7 - 3j/7 + 6k/7$
\item $2i/7 + 3j/7 - 6k/7$
\end{enumerate}

\item If in $\triangle ABC$, $BA = 2a$ and $BC = 3b$, then $AC$ is
\begin{enumerate}[label=(\Alph*)]
\item $2a + 3b$
\item $2a - 3b$
\item $3b - 2a$
\item $-2a - 3b$
\end{enumerate}

\item If $|a \times b| = \sqrt{3}$ and $a \cdot b = -3$, then the angle between $a$ and $b$ is
\begin{enumerate}[label=(\Alph*)]
\item $2\pi/3$
\item $\pi/6$
\item $\pi/3$
\item $5\pi/6$
\end{enumerate}

\newpage

\item Equation of the line passing through origin and making  
$30^\circ, 60^\circ, 90^\circ$ with $x,y,z$ axes respectively is
\begin{enumerate}[label=(\Alph*)]
\item $2x/\sqrt{3} = y/1 = z/0$
\item $2x/\sqrt{3} = 2y/1 = z/0$
\item $2x = 2y/\sqrt{3} = z/1$
\item $2x/\sqrt{3} = 2y/1 = z/1$
\end{enumerate}

\item If $P(A|B) = 2P(B|A)$ and $P(A) + P(B) = 2/3$, then $P(B)$ is
\begin{enumerate}[label=(\Alph*)]
\item $2/9$
\item $7/9$
\item $4/9$
\item $5/9$
\end{enumerate}

\item Anti-derivative of  
$(\tan x - 1)/(\tan x + 1)$ is
\begin{enumerate}[label=(\Alph*)]
\item $\sec^2(\pi/4 - x) + c$
\item $-\sec^2(\pi/4 - x) + c$
\item $\log|\sec(\pi/4 - x)| + c$
\item $-\log|\sec(\pi/4 - x)| + c$
\end{enumerate}

\item If $(a,b),(c,d),(e,f)$ are vertices of $\triangle ABC$ and $\Delta$ is its area, then
\[
\left|
\begin{matrix}
a & c & e\\
b & d & f\\
1 & 1 & 1
\end{matrix}
\right|^2
\]
is equal to
\begin{enumerate}[label=(\Alph*)]
\item $2\Delta^2$
\item $4\Delta^2$
\item $2\Delta$
\item $4\Delta$
\end{enumerate}

\item The function $f(x) = x|x|$ is
\begin{enumerate}[label=(\Alph*)]
\item continuous and differentiable at $x = 0$
\item continuous but not differentiable at $x = 0$
\item differentiable but not continuous at $x = 0$
\item neither differentiable nor continuous at $x = 0$
\end{enumerate}

\end{enumerate}
