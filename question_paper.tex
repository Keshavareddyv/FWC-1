\begin{enumerate}

\item Degree of the differential equation
$ \sin x + \cos x (dy/dx) - y^2 $
is
\begin{enumerate}
\item $2$
\item $1$
\item not defined
\item $0$
\end{enumerate}

\item The integrating factor of the differential equation
$ (1 - y^2)dx + xdy = 0 $
is
\begin{enumerate}
\item $1 - y^2$
\item $1/(1 - y^2)$
\item $x$
\item $1/x$
\end{enumerate}

\item Unit vector along vector PQ where
$ P(2,1,-1) $ and $ Q(4,-4,7) $
is
\begin{enumerate}
\item $2i - 5j + 8k$
\item $-2i + 5j - 8k$
\item $i + j + k$
\item $i - j + k$
\end{enumerate}

\item If in triangle ABC,
$ AB = 2a $ and $ BC = 3b $,
then $ AC $ is
\begin{enumerate}
\item $2a + 3b$
\item $2a - 3b$
\item $3b - 2a$
\item $a + b$
\end{enumerate}

\item If
$ |a \times b| = \sqrt{3} $
and
$ a b = 3 $,
then the angle between $ a $ and $ b $ is
\begin{enumerate}
\item $pi/6$
\item $pi/3$
\item $pi/2$
\item $2pi/3$
\end{enumerate}

\item Equation of the line passing through origin and making angles
$ 30 $, $ 60 $, $ 90 $
with x y z axes respectively is
\begin{enumerate}
\item $ x/\sqrt{3} = y = 0 $
\item $ \sqrt{3}x = y $
\item $ x + y + z = 0 $
\item $ x = y $
\end{enumerate}

\newpage

\item If
$ P(A) = 2P(B) $
and
$ P(A) + P(B) = 2/3 $,
then $ P(B) $ is
\begin{enumerate}
\item $2/9$
\item $1/9$
\item $4/9$
\item $1/3$
\end{enumerate}

\item Anti derivative of
$ 1/x $
is
\begin{enumerate}
\item $ \ln |x| + c $
\item $ x $
\item $ 1/x^2 $
\item $ e^x $
\end{enumerate}

\item If
$ a = (1,2,3) $
and
$ b = (2,3,4) $,
then $ |a \times b| $ is
\begin{enumerate}
\item $ \sqrt{3} $
\item $ \sqrt{6} $
\item $ \sqrt{12} $
\item $ 0 $
\end{enumerate}

\item The function
$ f(x) = |x| $
is
\begin{enumerate}
\item continuous and differentiable at $0$
\item continuous but not differentiable at $0$
\item differentiable but not continuous at $0$
\item neither continuous nor differentiable at $0$
\end{enumerate}

\end{enumerate}
